bash -c '
cd /workspace
if [ ! -d "fluxgym" ]; then git clone https://github.com/cocktailpeanut/fluxgym.git; else cd fluxgym && git pull && cd ..; fi
cd fluxgym
if [ ! -d "sd-scripts" ]; then git clone -b sd3 https://github.com/kohya-ss/sd-scripts.git; fi
python -m venv --system-site-packages env
source env/bin/activate
pip install -q --upgrade pip
cd sd-scripts && pip install -q --no-cache-dir -r requirements.txt && cd ..
pip install -q --no-cache-dir -r requirements.txt
pip install -q --no-cache-dir fastapi==0.116.1 starlette==0.47.2 uvicorn==0.35.0 httpcore==1.0.9 httpx==0.28.1 h11==0.16.0 anyio==4.9.0
export PYTHONWARNINGS="ignore"
nohup python3 app.py > fluxgym.log 2>&1 &
echo $! > fluxgym.pid
echo "FluxGym ready on port 7860"
tail -f fluxgym.log
'
